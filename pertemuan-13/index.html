ğŸ­ Virtual Function

class Kendaraan {
public:
    virtual void suara() {  // Virtual function
        cout << "Kendaraan bersuara" << endl;
    }
};

class Mobil : public Kendaraan {
public:
    void suara() override {  // Override
        cout << "Brum brum!" << endl;
    }
};

class Motor : public Kendaraan {
public:
    void suara() override {
        cout << "Ngeng ngeng!" << endl;
    }
};

    Kendaraan* ptr2 = &mo;

        cout << "Buku: " << judul << " oleh " << pengarang << endl;
    }
};
