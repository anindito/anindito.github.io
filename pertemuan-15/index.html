File Handling

Mengapa File Handling?

‚ùå Tanpa File Handling

  • Data hilang saat program ditutup
  • Harus input ulang setiap kali
  • Tidak ada persistensi data
  • Tidak bisa berbagi data

‚úÖ Dengan File Handling

  • Data tersimpan permanen
  • Bisa dimuat kembali
  • Persistensi data terjaga
  • Mudah dibagikan

File Teks vs File Binary

File Teks

  • Human-readable
  • Mudah di-debug
  • Platform-independent
  • Ukuran lebih besar

Gunakan untuk: Config files, logs, CSV, HTML

File Binary

  • Machine-readable
  • Lebih cepat
  • Ukuran lebih kecil
  • Efisien untuk struct

Gunakan untuk: Database, images, executables

Bagian 2

Exception Handling

Apa itu Exception?

Exception adalah kejadian tidak normal atau error yang terjadi saat program berjalan.

Contoh Exception:

  • Pembagian dengan nol
  • File tidak ditemukan
  • Array out of bounds
  • Memory allocation failed
  • Network timeout

Tanpa Exception Handling:

  • Program crash ‚ùå
  • Data hilang ‚ùå
  • User bingung ‚ùå
  • Sulit di-debug ‚ùå

Alur Exception Handling

Try-Catch Block

Struktur dasar exception handling
try {
    // Kode yang mungkin menghasilkan exception
    if (error_condition) {
        throw "Error message";
    }
    // Kode normal lainnya
}
catch (TipeException e) {
    // Kode untuk menangani exception
    cout << "Error: " << e << endl;
}
// Program tetap berjalan di sini

Contoh: Pembagian dengan Nol

#include <iostream>
using namespace std;

int main() {
    int a, b;
    cout << "Masukkan a: "; cin >> a;
    cout << "Masukkan b: "; cin >> b;
    
    try {
        if (b == 0) {
            throw "Error: Pembagian dengan nol!";
        }
        
        double hasil = static_cast<double>(a) / b;
        cout << "Hasil: " << hasil << endl;
    }
    catch (const char* pesan) {
        cout << pesan << endl;
    }
    
    cout << "Program selesai." << endl;
    return 0;
}

Multiple Catch Blocks

Menangani berbagai tipe exception
try {
    // Kode yang mungkin throw berbagai tipe exception
    if (condition1) throw 42;           // int
    if (condition2) throw 3.14;         // double
    if (condition3) throw "Error!";     // string
}
catch (int e) {
    cout << "Caught integer: " << e << endl;
}
catch (double e) {
    cout << "Caught double: " << e << endl;
}
catch (const char* e) {
    cout << "Caught string: " << e << endl;
}
catch (...) {
    // Catch-all: menangani exception apapun
    cout << "Caught unknown exception!" << endl;
}

Menggunakan Standard Exceptions

#include <stdexcept>
using namespace std;

double bagi(double a, double b) {
    if (b == 0) {
        throw invalid_argument("Pembagi tidak boleh nol!");
    }
    return a / b;
}

int akses(int arr[], int size, int index) {
    if (index < 0 || index >= size) {
        throw out_of_range("Indeks di luar jangkauan!");
    }
    return arr[index];
}

int main() {
    try {
        double hasil = bagi(10, 0);
    }
    catch (invalid_argument& e) {
        cout << "Error: " << e.what() << endl;
    }
    return 0;
}

Exception dalam File Operations

#include <fstream>
#include <stdexcept>

void bacaFile(string namaFile) {
    ifstream file(namaFile);
    
    if (!file.is_open()) {
        throw runtime_error("Tidak dapat membuka: " + namaFile);
    }
    
    string baris;
    while (getline(file, baris)) {
        cout << baris << endl;
    }
    
    if (file.bad()) {
        throw runtime_error("Error saat membaca file!");
    }
    
    file.close();
}

int main() {
    try {
        bacaFile("data.txt");
    }
    catch (runtime_error& e) {
        cout << "Error: " << e.what() << endl;
    }
}

Custom Exception Classes

class SaldoTidakCukupException : public exception {
private:
    string pesan;
public:
    SaldoTidakCukupException(double saldo, double penarikan) {
        pesan = "Saldo tidak cukup! Saldo: " + 
                to_string(saldo) + 
                ", Penarikan: " + to_string(penarikan);
    }
    
    const char* what() const noexcept override {
        return pesan.c_str();
    }
};

// Penggunaan:
try {
    if (penarikan > saldo) {
        throw SaldoTidakCukupException(saldo, penarikan);
    }
}
catch (SaldoTidakCukupException& e) {
    cout << e.what() << endl;
}

Best Practices Exception Handling

‚úÖ DO

  • Catch by reference
  • Catch specific first
  • Use standard exceptions
  • Throw by value
  • Use RAII pattern
  • Provide clear messages

‚ùå DON'T

  • Catch by value
  • Throw in destructor
  • Catch and ignore
  • Use exceptions for control flow
  • Throw pointer
  • Empty catch blocks

RAII Pattern

RAII = Resource Acquisition Is Initialization

Resource diperoleh di constructor, dibebaskan di destructor

‚ùå Buruk: Manual cleanup
void proses() {
    ifstream* file = 
        new ifstream("data.txt");
    
    // Jika exception terjadi,
    // file tidak tertutup!
    
    delete file;
}
‚úÖ Baik: RAII
void proses() {
    ifstream file("data.txt");
    
    // Jika exception terjadi,
    // destructor otomatis
    // menutup file!
    
} // Otomatis cleanup

Custom Exceptions

class BukuTidakDitemukanException : public exception {
private:
    string isbn;
public:
    BukuTidakDitemukanException(string i) : isbn(i) {}
    const char* what() const noexcept override {
        return "Buku tidak ditemukan!";
    }
    string getISBN() const { return isbn; }
};

class BukuSudahDipinjamException : public exception {
public:
    const char* what() const noexcept override {
        return "Buku sudah dipinjam!";
    }
};

Method: Muat dari File

void muatDariFile() {
    ifstream file(namaFile, ios::binary);
    
    if (!file.is_open()) {
        return; // File belum ada
    }
    
    Buku buku;
    while (file.read(reinterpret_cast<char*>(&buku), 
                     sizeof(Buku))) {
        daftarBuku.push_back(buku);
    }
    
    if (file.bad()) {
        throw runtime_error("Error saat membaca database!");
    }
    
    file.close();
}

Method: Simpan ke File

void simpanKeFile() {
    ofstream file(namaFile, ios::binary | ios::trunc);
    
    if (!file.is_open()) {
        throw runtime_error("Tidak dapat menyimpan database!");
    }
    
    for (const auto& buku : daftarBuku) {
        file.write(reinterpret_cast<const char*>(&buku), 
                   sizeof(Buku));
        
        if (file.fail()) {
            throw runtime_error("Error saat menulis!");
        }
    }
    
    file.close();
}

Method: Pinjam Buku

void pinjamBuku(string isbn) {
    for (auto& buku : daftarBuku) {
        if (buku.isbn == isbn) {
            if (buku.dipinjam) {
                throw BukuSudahDipinjamException();
            }
            buku.dipinjam = true;
            cout << "Buku '" << buku.judul 
                 << "' berhasil dipinjam!" << endl;
            return;
        }
    }
    throw BukuTidakDitemukanException(isbn);
}

Main Program dengan Exception Handling

int main() {
    try {
        Perpustakaan perpus("perpustakaan.dat");
        
        // Menu loop...
        try {
            perpus.pinjamBuku("ISBN-001");
        }
        catch (BukuTidakDitemukanException& e) {
            cout << "Error: " << e.what() << endl;
            cout << "ISBN: " << e.getISBN() << endl;
        }
        catch (BukuSudahDipinjamException& e) {
            cout << "Error: " << e.what() << endl;
        }
    }
    catch (exception& e) {
        cout << "Fatal Error: " << e.what() << endl;
        return 1;
    }
    
    return 0;
}
