Membaca Input & Serial

const int BUTTON_PIN = 2;
const int LED_PIN = 13;

void setup() {
    pinMode(BUTTON_PIN, INPUT_PULLUP);
    pinMode(LED_PIN, OUTPUT);
    Serial.begin(9600);  // Komunikasi serial
    Serial.println("Sistem siap!");
}

void loop() {
    int state = digitalRead(BUTTON_PIN);
    
    if (state == LOW) {  // Tombol ditekan
        digitalWrite(LED_PIN, HIGH);
        Serial.println("Tombol ditekan!");
    } else {
        digitalWrite(LED_PIN, LOW);
    }
    delay(100);
}

Sensor Analog (LM35)

const int TEMP_PIN = A0;

void setup() {
    Serial.begin(9600);
    Serial.println("Monitoring Suhu");
}

void loop() {
    int nilai = analogRead(TEMP_PIN);  // 0-1023
    
    // Konversi ke tegangan (0-5V)
    float tegangan = nilai * (5.0 / 1023.0);
    
    // LM35: 10mV per derajat Celcius
    float suhu = tegangan * 100.0;
    
    Serial.print("Suhu: ");
    Serial.print(suhu);
    Serial.println(" C");
    
    delay(1000);
}

Binary Search

int binarySearch(vector<int>& arr, int target) {
    int left = 0, right = arr.size() - 1;
    
    while (left <= right) {
        int mid = left + (right - left) / 2;  // Hindari overflow
        
        if (arr[mid] == target) {
            return mid;  // Ditemukan!
        } else if (arr[mid] < target) {
            left = mid + 1;   // Cari di kanan
        } else {
            right = mid - 1;  // Cari di kiri
        }
    }
    
    return -1;  // Tidak ditemukan
}
// Kompleksitas: O(log n)

SFML Example

// Game sederhana dengan SFML
#include <SFML/Graphics.hpp>

int main() {
    sf::RenderWindow window(sf::VideoMode(800, 600), "Game");
    sf::CircleShape player(25.f);
    player.setFillColor(sf::Color::Green);
    player.setPosition(400.f, 300.f);
    
    while (window.isOpen()) {
        sf::Event event;
        while (window.pollEvent(event)) {
            if (event.type == sf::Event::Closed)
                window.close();
        }
        
        // Movement
        if (sf::Keyboard::isKeyPressed(sf::Keyboard::W))
            player.move(0, -5);
            
        window.clear();
        window.draw(player);
        window.display();
    }
}

AABB Collision Detection

struct Rectangle {
    float x, y, width, height;
};

bool checkCollision(const Rectangle& a, const Rectangle& b) {
    return (a.x < b.x + b.width &&
            a.x + a.width > b.x &&
            a.y < b.y + b.height &&
            a.y + a.height > b.y);
}

// Penggunaan
Rectangle player = {100, 100, 50, 50};
Rectangle enemy = {120, 110, 40, 40};

if (checkCollision(player, enemy)) {
    cout << "TABRAKAN!" << endl;
}

Review: Exception Handling

#include <stdexcept>

double bagi(int a, int b) {
    if (b == 0) {
        throw runtime_error("Pembagian dengan nol!");
    }
    return (double)a / b;
}

int main() {
    try {
        double hasil = bagi(10, 0);
        cout << hasil << endl;
    }
    catch (const runtime_error& e) {
        cerr << "Error: " << e.what() << endl;
    }
    catch (...) {
        cerr << "Unknown error" << endl;
    }
    return 0;
}

  1. Baca soal dengan teliti - Pahami requirement

  4. Implementasi - Tulis kode
  5. Test dengan contoh - Trace manual
  6. Review - Cek edge cases
