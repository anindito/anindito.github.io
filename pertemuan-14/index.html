üìñ Membaca dengan Operator >>


#include <iostream>
#include <fstream>
using namespace std;

int main() {
    ifstream file("angka.txt");
    
    int angka;
    int total = 0;
    
    // Baca semua angka sampai EOF
    while (file >> angka) {
        cout << "Baca: " << angka << endl;
        total += angka;
    }
    
    cout << "Total: " << total << endl;
    file.close();
    
    return 0;
}

Operator >> membaca hingga whitespace

üîÑ Alur Exception Handling

üéØ Try - Catch - Throw

try

Blok kode yang mungkin error

try {
  // kode berisiko
}

throw

Lempar exception

throw "Error!";
throw 404;

catch

Tangkap & tangani

catch (tipe e) {
  // handle
}

üíª Contoh Exception Handling


#include <iostream>
using namespace std;

int main() {
    int a = 10, b = 0;
    
    try {
        if (b == 0) {
            throw "Tidak bisa bagi dengan nol!";
        }
        int hasil = a / b;
        cout << "Hasil: " << hasil << endl;
    }
    catch (const char* pesan) {
        cout << "Error: " << pesan << endl;
    }
    
    cout << "Program tetap jalan!" << endl;
    return 0;
}
‚úÖ Program tidak crash, error tertangani dengan baik!

üé£ Multiple Catch Blocks


try {
    // kode yang bisa throw berbagai tipe exception
}
catch (const runtime_error& e) {
    cout << "Runtime Error: " << e.what() << endl;
}
catch (const invalid_argument& e) {
    cout << "Invalid Argument: " << e.what() << endl;
}
catch (int kode) {
    cout << "Error code: " << kode << endl;
}
catch (...) {
    cout << "Unknown error!" << endl;
}

catch(...) menangkap semua exception

üîó Exception + File Handling


#include <iostream>
#include <fstream>
#include <stdexcept>
using namespace std;

void bacaFile(const string& nama) {
    ifstream file(nama);
    
    if (!file.is_open()) {
        throw runtime_error("Gagal buka: " + nama);
    }
    
    string baris;
    while (getline(file, baris)) {
        cout << baris << endl;
    }
    file.close();
}

int main() {
    try {
        bacaFile("tidak_ada.txt");
    }
    catch (const runtime_error& e) {
        cout << "Error: " << e.what() << endl;
    }
    return 0;
}

‚úÖ Best Practice #1

Defensive Programming

"Never trust input" - Selalu validasi data!


// Fungsi baca integer dengan validasi
int bacaInteger(int min, int max) {
    int nilai;
    while (true) {
        if (cin >> nilai && nilai >= min && nilai <= max) {
            return nilai;
        }
        cout << "Input tidak valid! Coba lagi." << endl;
        cin.clear();
        cin.ignore(10000, '\n');
    }
}

‚úÖ Best Practice #2

RAII Pattern

RAII = Resource Acquisition Is Initialization

Resource otomatis dibersihkan saat objek dihancurkan


void prosesFile() {
    ofstream file("data.txt");  // Resource acquired
    
    file << "Data 1" << endl;
    file << "Data 2" << endl;
    
    // Meskipun ada exception, file tetap ditutup
    // karena destructor dipanggil otomatis
}  // Resource released (file.close() otomatis)

üì± Contoh: Buku Alamat


class BukuAlamat {
public:
    void tambah(const Kontak& k) {
        if (k.nama.empty()) {
            throw invalid_argument("Nama kosong!");
        }
        daftar.push_back(k);
    }
    
    void simpan(const string& file) {
        ofstream f(file);
        if (!f.is_open()) {
            throw runtime_error("Gagal simpan!");
        }
        for (auto& k : daftar) {
            f << k.nama << "|" << k.telp << endl;
        }
    }
};

int main() {
    try {
        BukuAlamat buku;
        buku.tambah({"Ahmad", "08123456789"});
        buku.simpan("kontak.txt");
    } catch (const exception& e) {
        cout << "Error: " << e.what() << endl;
    }
}
