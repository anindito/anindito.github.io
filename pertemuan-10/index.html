Mengapa Harus Menentukan Ukuran Kolom?

Compiler perlu tahu ukuran kolom untuk menghitung offset memori

Formula: alamat = base + (i Ã— jumlah_kolom + j) Ã— sizeof(tipe)

Tanpa ukuran kolom, compiler tidak bisa menghitung posisi elemen!

Fungsi untuk Operasi Array 2D

const int MAX = 10;

int cariMaksimum(int arr[][MAX], int baris, int kolom) {
    int maks = arr[0][0];
    for (int i = 0; i < baris; i++) {
        for (int j = 0; j < kolom; j++) {
            if (arr[i][j] > maks) {
                maks = arr[i][j];
            }
        }
    }
    return maks;
}

double hitungRataRata(int arr[][MAX], int baris, int kolom) {
    int jumlah = 0;
    for (int i = 0; i < baris; i++) {
        for (int j = 0; j < kolom; j++) {
            jumlah += arr[i][j];
        }
    }
    return (double)jumlah / (baris * kolom);
}

void tambahMatrix(int A[][MAX], int B[][MAX], 
                  int C[][MAX], int baris, int kolom) {
    for (int i = 0; i < baris; i++) {
        for (int j = 0; j < kolom; j++) {

Perkalian Matriks

A (mÃ—n) Ã— B (nÃ—p) = C (mÃ—p)

void kaliMatrix(int A[][MAX], int B[][MAX], int C[][MAX],
                int m, int n, int p) {
    // Inisialisasi C dengan 0
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < p; j++) {
            C[i][j] = 0;
        }
    }
    
    // Perkalian
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < p; j++) {
            for (int k = 0; k < n; k++) {
                C[i][j] += A[i][k] * B[k][j];
            }
        }
    }
}

void transpose(int A[][MAX], int T[][MAX], 
               int baris, int kolom) {
    for (int i = 0; i < baris; i++) {
        for (int j = 0; j < kolom; j++) {
            T[j][i] = A[i][j];
        }
    }
}

void buatIdentitas(int I[][MAX], int n) {
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            I[i][j] = (i == j) ? 1 : 0;

    }
